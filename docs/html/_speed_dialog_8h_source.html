54 void snapToPosition() {
55 // 计算最近的十分之一位置
56 int value = this->value();
57 int min = this->minimum();
58 int max = this->maximum();
59
60 // 计算步长
61 double step = (max - min) / 10.0;
62
63 // 找到最近的十分之一点
64 int closestPoint = min;
65 double minDistance = std::numeric_limits<double>::max();
66
67 for (int i = 0; i <= 10; i++) {
68 int point = min + qRound(i * step);
69 double distance = std::abs(value - point);
70
71 if (distance < minDistance) {
72 minDistance = distance;
73 closestPoint = point;
74 }
75 }
76 if (closestPoint == minimum())closestPoint = minimum() + 1;
77 // 直接设置到最近的点（跳过中间值）
78 setSliderPosition(closestPoint);
79 emit numChanged(closestPoint / 10);
80 }

88 void mousePressEvent(QMouseEvent *event) override {
89 // 先调用基类处理

91 if (event->button() == Qt::LeftButton) {
92 setCursor(Qt::ClosedHandCursor); // 按下时变为闭合手型

102 void mouseMoveEvent(QMouseEvent *event) override {
103 // 先调用基类处理（更新位置）
104 QtMaterialSlider::mouseMoveEvent(event);
105 if (event->buttons() & Qt::LeftButton) {
106 // 拖拽时保持闭合手型
107 setCursor(Qt::ClosedHandCursor);
108 }
109 if (!m_isPress)return;

126 }
