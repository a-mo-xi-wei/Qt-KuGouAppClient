= [](QString input) {
QString result = input;
int length = input.length();
ushort xor_code = length % 255;
for (int i = 0; i < length; ++i)
{
result[i] = QChar(static_cast<ushort>(input[i].unicode() ^ xor_code));
}
return result;
}
