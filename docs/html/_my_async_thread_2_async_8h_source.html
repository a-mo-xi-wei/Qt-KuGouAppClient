371 template <typename ResultType, typename Function, typename... Args>
372 class AsyncJob : public QRunnable
373 {
374 public:
383 AsyncJob(Function &&function, Args&&... args)
384 // decay copy like std::thread
385 : data(decayCopy(std::forward<Function>(function)), decayCopy(std::forward<Args>(args))...)
386 {
387 // we need to report it as started even though it isn't yet, because someone might
388 // call waitForFinished on the future, which does _not_ block if the future is not started
389 futureInterface.setRunnable(this);
390 futureInterface.reportStarted();
391 }
392
398 ~AsyncJob()
399 {
400 // QThreadPool can delete runnables even if they were never run (e.g. QThreadPool::clear).
401 // Since we reported them as started, we make sure that we always report them as finished.
402 // reportFinished only actually sends the signal if it wasn't already finished.
403 futureInterface.reportFinished();
404 }
405
411 QFuture<ResultType> future() { return futureInterface.future(); }
412
418 void run() override
419 {
420 if (priority != QThread::InheritPriority)
421 if (QThread *thread = QThread::currentThread())
422 if (thread != qApp->thread())
423 thread->setPriority(priority);
424 if (futureInterface.isCanceled()) {
425 futureInterface.reportFinished();
426 return;
427 }
428 runHelper(std::make_index_sequence<std::tuple_size<Data>::value>());
429 }
430
436 void setThreadPool(QThreadPool *pool)
437 {
438 futureInterface.setThreadPool(pool);
439 }
440
446 void setThreadPriority(QThread::Priority p)
447 {
448 priority = p;
449 }
450
451 private:
455 using Data = std::tuple<std::decay_t<Function>, std::decay_t<Args>...>;
456
462 template <std::size_t... index>
463 void runHelper(std::index_sequence<index...>)
464 {
465 // invalidates data, which is moved into the call
466 runAsyncImpl(futureInterface, std::move(std::get<index>(data))...);
467 if (futureInterface.isSuspended())
468 futureInterface.waitForResume();
469 futureInterface.reportFinished();
470 }

475 Data data;

480 QFutureInterface<ResultType> futureInterface;

485 QThread::Priority priority = QThread::InheritPriority;

687 template <typename Function, typename... Args,
688 typename ResultType = typename resultType<Function>::type>
689 QFuture<ResultType>
690 runAsync(QThreadPool *pool, QThread::Priority priority, Function &&function, Args&&... args)
691 {
692 auto job = new AsyncJob<ResultType, Function, Args...>
693 (std::forward<Function>(function), std::forward<Args>(args)...);
694 job->setThreadPriority(priority);
695 QFuture<ResultType> future = job->future();
696 if (pool) {
697 job->setThreadPool(pool);
698 pool->start(job);
699 } else {
700 auto thread = new RunnableThread(job);
701 thread->moveToThread(qApp->thread()); // make sure thread gets deleteLater on main thread
702 QObject::connect(thread, &QThread::finished, thread, &QObject::deleteLater);
703 thread->start(priority);
704 }
705 return future;
706 }
