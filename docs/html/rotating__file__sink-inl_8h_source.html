26template <typename Mutex>
27SPDLOG_INLINE rotating_file_sink<Mutex>::rotating_file_sink(
28 filename_t base_filename,
29 std::size_t max_size,
30 std::size_t max_files,
31 bool rotate_on_open,
32 const file_event_handlers &event_handlers)
33 : base_filename_(std::move(base_filename)),
34 max_size_(max_size),
35 max_files_(max_files),
36 file_helper_{event_handlers} {
37 if (max_size == 0) {
38 throw_spdlog_ex("rotating sink constructor: max_size arg cannot be zero");
39 }
40
41 if (max_files > 200000) {
42 throw_spdlog_ex("rotating sink constructor: max_files arg cannot exceed 200000");
43 }
44 file_helper_.open(calc_filename(base_filename_, 0));
45 current_size_ = file_helper_.size(); // expensive. called only once
46 if (rotate_on_open && current_size_ > 0) {
47 rotate_();
48 current_size_ = 0;
49 }
50}

52// calc filename according to index and file extension if exists.
53// e.g. calc_filename("logs/mylog.txt, 3) => "logs/mylog.3.txt".
54template <typename Mutex>
55SPDLOG_INLINE filename_t rotating_file_sink<Mutex>::calc_filename(const filename_t &filename,
56 std::size_t index) {
57 if (index == 0u) {
58 return filename;
59 }
60
61 filename_t basename, ext;
62 std::tie(basename, ext) = details::file_helper::split_by_extension(filename);
63 return fmt_lib::format(SPDLOG_FMT_STRING(SPDLOG_FILENAME_T("{}.{}{}")), basename, index, ext);
64}

78template <typename Mutex>
79SPDLOG_INLINE void rotating_file_sink<Mutex>::sink_it_(const details::log_msg &msg) {
80 memory_buf_t formatted;
81 base_sink<Mutex>::formatter_->format(msg, formatted);
82 auto new_size = current_size_ + formatted.size();
83
84 // rotate if the new estimated file size exceeds max size.
85 // rotate only if the real size > 0 to better deal with full disk (see issue #2261).
86 // we only check the real size when new_size > max_size_ because it is relatively expensive.
87 if (new_size > max_size_) {
88 file_helper_.flush();
89 if (file_helper_.size() > 0) {
90 rotate_();
91 new_size = formatted.size();
92 }
93 }
94 file_helper_.write(formatted);
95 current_size_ = new_size;
96}

103// Rotate files:
104// log.txt -> log.1.txt
105// log.1.txt -> log.2.txt
106// log.2.txt -> log.3.txt
107// log.3.txt -> delete
108template <typename Mutex>
109SPDLOG_INLINE void rotating_file_sink<Mutex>::rotate_() {
110 using details::os::filename_to_str;
111 using details::os::path_exists;
112
113 file_helper_.close();
114 for (auto i = max_files_; i > 0; --i) {
115 filename_t src = calc_filename(base_filename_, i - 1);
116 if (!path_exists(src)) {
117 continue;
118 }
119 filename_t target = calc_filename(base_filename_, i);
120
121 if (!rename_file_(src, target)) {
122 // if failed try again after a small delay.
123 // this is a workaround to a windows issue, where very high rotation
124 // rates can cause the rename to fail with permission denied (because of antivirus?).
125 details::os::sleep_for_millis(100);
126 if (!rename_file_(src, target)) {
127 file_helper_.reopen(
128 true); // truncate the log file anyway to prevent it to grow beyond its limit!
129 current_size_ = 0;
130 throw_spdlog_ex("rotating_file_sink: failed renaming " + filename_to_str(src) +
131 " to " + filename_to_str(target),
132 errno);
133 }
134 }
135 }
136 file_helper_.reopen(true);
137}

139// delete the target if exists, and rename the src file to target
140// return true on success, false otherwise.
141template <typename Mutex>
142SPDLOG_INLINE bool rotating_file_sink<Mutex>::rename_file_(const filename_t &src_filename,
143 const filename_t &target_filename) {
144 // try to delete the target file in case it already exists.
145 (void)details::os::remove(target_filename);
146 return details::os::rename(src_filename, target_filename) == 0;
147}
