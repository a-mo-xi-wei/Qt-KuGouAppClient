435static inline void itimeofday(long *sec, long *usec)
436{
437#if defined(__unix)
438 struct timeval time;
439 gettimeofday(&time, NULL);
440 if (sec) *sec = time.tv_sec;
441 if (usec) *usec = time.tv_usec;
442#else
443 static long mode = 0, addsec = 0;
444 BOOL retval;
445 static long long freq = 1;
446 long long qpc;
447 if (mode == 0) {
448 retval = QueryPerformanceFrequency((LARGE_INTEGER*)&freq);
449 freq = (freq == 0) ? 1 : freq;
450 retval = QueryPerformanceCounter((LARGE_INTEGER*)&qpc);
451 addsec = (long)time(NULL);
452 addsec = addsec - (long)((qpc / freq) & 0x7fffffff);
453 mode = 1;
454 }
455 retval = QueryPerformanceCounter((LARGE_INTEGER*)&qpc);
456 retval = retval * 2;
457 if (sec) *sec = (long)(qpc / freq) + addsec;
458 if (usec) *usec = (long)((qpc % freq) * 1000000 / freq);
459#endif
460}
